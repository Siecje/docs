buttons
=============

@short:
	sets a custom template for buttons or hides them

@type: string|boolean
@example:
webix.ui({
	view:"dbllist",
    buttons:false
});

//or

webix.ui({
	view:"dbllist",
    buttons:"<button class='dbllist_button'>Select</button>"
});

@template:	api_config
