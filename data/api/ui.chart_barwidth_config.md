barWidth
=============


@short:
	the width of bars

@type: number
@example:
webix.ui({
	view:"chart",
	type:"bar",
	value:"#sales#",
	label:"#sales#",
	barWidth:35,
    data: dataset
});

@template:	api_config
@defined:	ui.chart	
@relatedsample:
	08_chart/06_bar_chart/01_init.html
@related:
