subRender
=============

@short:
	defines if the sub-elements should be rendered for the first-level branches

@type: boolean
@default:true
@example:
webix.ui({
    view:"treemap",
    subRender: false,
    url: "data/data.json"
});

@template:	api_config
@descr:
