separator
=============


@short:
	a delimiter for separating options in a multicombo, comma by default

@type: string
@example:
{
	view:"multicombo", 
    label:"Name", 
    labelPosition: "top",
	value:"1|7|9|12",
	separator: "|", //data contains commas
	...
}
