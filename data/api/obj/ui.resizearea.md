### Constructor

~~~js
	var resizearea = webix.ui({
		view:"resizearea", 
		container:"mydiv", 
		...config options goes here..
	})
	//or, in case of jQuery
	$("#mydiv").webix_resizearea({
		...config options goes here..
	});
~~~
