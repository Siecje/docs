getValues
=============

@short: gets data hash
	
@params:

@example:

webix.ui({
	view:"template",
    id:"template1",
    data:{
    	src:"imgs/image002.jpg"
    }
});

var values = $$("template1").getValues(); //{src:"imgs/image002.jpg"}
