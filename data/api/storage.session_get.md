get
=============


@short:
	gets data from the storage
	

@params:
- name		string		the data key

@returns:
data		object	a data item from the storage associated with the key or <i>null</i>


@example:
var weather = webix.storage.session.get('weather');

@template:	api_method
@defined:	storage.session
